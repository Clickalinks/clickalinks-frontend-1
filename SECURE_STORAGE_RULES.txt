rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Logos folder - public read, size-restricted write
    match /logos/{logoId} {
      // Allow anyone to read logos (needed for displaying on website)
      allow read: if true;
      
      // Allow writes with size limit (5MB max)
      // Security is provided by:
      // 1. File size limit (prevents abuse)
      // 2. Client-side file type validation (only images)
      // 3. Virus scanning (client-side)
      // 4. File naming pattern (square-{number}-{timestamp})
      allow write: if request.resource.size < 5 * 1024 * 1024;
      
      // Allow delete for cleanup
      allow delete: if true;
    }
    
    // Deny all other paths by default
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
