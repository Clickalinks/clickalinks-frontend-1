# Firestore Security Rules
# Copy these rules to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write access to purchasedSquares collection
    match /purchasedSquares/{squareId} {
      allow read: if true; // Public read access
      allow write: if true; // Allow writes (you may want to restrict this in production)
    }
    
    // Default: deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

# IMPORTANT: 
# 1. Go to Firebase Console: https://console.firebase.google.com/
# 2. Select your project: clickalinks-frontend
# 3. Go to Firestore Database > Rules tab
# 4. Replace the existing rules with the rules above
# 5. Click "Publish"
